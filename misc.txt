    >> python setup.py bdist_wheel -d .
    >> python setup-nodeps.py bdist_wheel -d .
    Publish pypi package cmd:
    >> twine upload lpd-0.3.7-py3-none-any.whl
    >> twine upload lpd_nodeps-0.3.7-py3-none-any.whl
